139 {
140  int tamanhoEmBytes = variavel->obterTamanhoMaximoEmBytes();
141  auto cursorAntes = dataInputStream.obterCursor();
142 
143  variavel->lerBytes(dataInputStream);
144 
145  auto diferenca = dataInputStream.obterCursor() - cursorAntes;
146 
147  // Checa se a entidade não consumiu todos os bytes a que tem direito
148  if (diferenca < tamanhoEmBytes)
149  {
150  dataInputStream.moverCursor(tamanhoEmBytes - diferenca);
151  }
152 
153  else if (diferenca > tamanhoEmBytes)
154  {
155  cerr << "[Serializavel]: Algumas de suas entidades leu mais bytes ("

157  << tamanhoEmBytes << ")." << endl << "Este é o vetor de bytes que ela"

159  }
