355  void promoverElementoPara(Pagina* paginaDestino, int indice, Pagina* exPaginaCheia,
356  Pagina* novaPagina, bool promoverElementoDoFim, fstream& arquivo)
357  {
358  // Checa se a ex página cheia será afetada na promoção
359  if (promoverElementoDoFim)
360  {
361  exPaginaCheia->ponteiros.erase(
362  exPaginaCheia->ponteiros.begin() + exPaginaCheia->numeroDeElementos);
363 
364  transferirElementoPara(paginaDestino, indice, numeroDeElementos - 1);
365  }
366 
367  else // A página nova será afetada na promoção
368  {
369  novaPagina->ponteiros.erase(novaPagina->ponteiros.begin());
370 
371  transferirElementoPara(paginaDestino, indice, 0);
372  }
373 
374  numeroDeElementos--;
375 
376  // atualiza as páginas no arquivo
377  exPaginaCheia->colocarNoArquivo(arquivo);
378  novaPagina->colocarNoArquivo(arquivo);
379 
380  // Quando um elemento é promovido, o ponteiro da esquerda dele deve apontar
381  // para a página que estava cheia (a que provocou a duplicação).
382  // Já o ponteiro da direita deve apontar para a nova página (gerada pela
383  // duplicação).
384  paginaDestino->ponteiros[indice] = exPaginaCheia->endereco;
385  paginaDestino->ponteiros[indice + 1] = novaPagina->endereco;
386  paginaDestino->colocarNoArquivo(arquivo);
387  }
