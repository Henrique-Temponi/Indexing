533  {
534  TIPO_DOS_DADOS dadoExcluido;

539  // Checa se a chave realmente foi encontrada
540  if (paginaFilha->chaves[indiceDaChave] == chave)
541  {
542  dadoExcluido = paginaFilha->dados[indiceDaChave];
543  limparErro();

546  {
547  paginaFilha->excluir(indiceDaChave, false, true);
548 
549  // Checa se o tamanho da paginaFilha antes da remoção era <= a 50%
550  if (paginaFilha->tamanho() + 1 <= numeroDeChavesPorPagina / 2)
551  {

563  {
564  swap(paginaPai, paginaFilha);
565  pilhaDeIndices.pop_back();
566  pilhaDeEnderecos.pop_back();
567  carregar(paginaPai, pilhaDeEnderecos.back());
568  fundirPaginas(pilhaDeIndices.back());
569  }
570

572  trocarRaizPor(paginaFilha);

577  }
578 
579  else

581  trocarChavePorAntecessora(
582  indiceDaChave, pilhaDeEnderecos, pilhaDeIndices);
583  excluir(chave, pilhaDeEnderecos, pilhaDeIndices);

586

588

707  void inserir(TIPO_DAS_CHAVES &chave, TIPO_DOS_DADOS &dado,
708  list<file_ptr_type>& pilhaDeEnderecos,
709  list<int>& pilhaDeIndices)
710  {
711  // Após a função obterCaminhoDeDescida(), paginaFilha aponta para a última
712  // página carregada na descida da árvore (alguma folha).
713  int indiceDeInsercao = paginaFilha->obterIndiceDeDescida(chave);
714 
715  // Checa se a inserção teve sucesso
716  if (paginaFilha->inserir(chave, dado, indiceDeInsercao))
717  paginaFilha->colocarNoArquivo(arquivo); // Encerra salvando a página
718 
719  else // Inserção na página falhou, acontece quando ela está cheia.
720  {
721  pilhaDeEnderecos.pop_back(); // O último endereço pode ser descartado
722  bool voltouParaARaiz = pilhaDeEnderecos.empty();
723  int indiceDePromocao = pilhaDeIndices.back();
724 
725  pair<Pagina *, bool> infoPai(nullptr, false);
726  tratarPaginaCheia(chave, dado, indiceDePromocao, infoPai);
727 
728  if (voltouParaARaiz) trocarRaizPor(paginaPai);
729 
730  else // Inicia o processo de subida na árvore
731  {
732  // Quando uma página que é filha da atual promove um elemento para
733  // a atual ou para a irmã da atual, a variável paginaDeInsercao vai
734  // ter um ponteiro para a página que receber o elemento promovido.
735  Pagina *paginaDeInsercao = infoPai.first;
736  bool inseriuNaPaginaAtual;
737  file_ptr_type enderecoPaginaPai;
738 
739  // Checa se houve alguma promoção para a página atual
740  while (paginaDeInsercao != nullptr)
741  {
742  pilhaDeEnderecos.pop_back();
743  voltouParaARaiz = pilhaDeEnderecos.empty();
744  enderecoPaginaPai = voltouParaARaiz ?
745  constantes::ptrNuloPagina : pilhaDeEnderecos.back();
746 
747  pilhaDeIndices.pop_back();
748  indiceDePromocao = pilhaDeIndices.back();
749  inseriuNaPaginaAtual = infoPai.second;
750 
751  // As páginas que eram pais, na subida da árvore, são filhas
752  swap(paginaPai, paginaFilha);
753  swap(paginaIrmaPai, paginaIrma);
754 
755  if (voltouParaARaiz) // Já está na raiz e precisa-se promover
756  {
757  paginaPai->limpar(); // Cria a nova raiz
758  paginaPai->ponteiros.push_back(constantes::ptrNuloPagina);
759  }
760

762 
763  promoverOParQueEstiverSobrando(

765  inseriuNaPaginaAtual, infoPai);
766

770  }
