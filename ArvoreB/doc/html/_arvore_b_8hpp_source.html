323  pair<Pagina *, bool> inserir(
324  TIPO_DAS_CHAVES chave, TIPO_DOS_DADOS dado,
325  int indiceDePromocao, file_pointer_type enderecoPaginaFilha,
326  file_pointer_type enderecoPaginaPai)
327  {
328  pair<Pagina *, bool> infoPai(nullptr, false);
329 
330  // Checa se a página foi carregada
331  if (carregar(paginaFilha, enderecoPaginaFilha))
332  {
333  // Checa se não existe um endereço para a página pai. Quando ele
334  // não existir é porque a recursividade voltou à raiz.
335  bool voltouParaARaiz =
336  enderecoPaginaPai == constantes::ptrNuloPagina;
337 
338  int indiceDeInsercao = paginaFilha->obterIndiceDeDescida(chave);
339  file_pointer_type ponteiroDeDescida =
340  paginaFilha->ponteiros[indiceDeInsercao];
341 
342  // Checa se não há ponteiro de descida. Caso não, a página é uma folha.
343  if (ponteiroDeDescida == constantes::ptrNuloPagina)
344  {
345  // Checa se a inserção teve sucesso
346  if (paginaFilha->inserir(chave, dado, indiceDeInsercao))
347  {
348  paginaFilha->colocarNoArquivo(arquivo);
349  }
350 
351  else // Inserção na página falhou, acontece quando ela está cheia.
352  {
353  tratarPaginaCheia(chave, dado, indiceDePromocao, infoPai);
354 
355  if (voltouParaARaiz) trocarRaizPor(paginaPai);
356  }
357  }
358 
359  else // A página não é uma folha, continuar descendo
360  {
361  // O swap é necessário pois cada um desses ponteiros aponta para um
362  // objeto página concreto e a referência para este objeto não pode
363  // ser perdida
364  swap(paginaFilha, paginaPai); // A página filha passar a ser pai
365 
366  // O inserir retorna informações sobre a promoção de elementos para
367  // esta página
368  infoPai = inserir(
369  chave, dado, indiceDeInsercao,
370  ponteiroDeDescida, enderecoPaginaFilha);
371 
372  // Checa se esta página precisa promover algum elemento que tenha
373  // sido adicionado nela pelas suas filhas
374  if (infoPai.first != nullptr)
375  {
376  Pagina *paginaDeInsercao = infoPai.first;
377  bool inseriuNaPaginaFilha = infoPai.second;
378 
379  // As páginas que eram pais, na volta da recursividade, são filhas
380  swap(paginaPai, paginaFilha);
381  swap(paginaIrmaPai, paginaIrma);
382 
383  if (voltouParaARaiz)
384  {
385  paginaPai->limpar();
386  paginaPai->ponteiros.push_back(constantes::ptrNuloPagina);
387  }
388 
389  else carregar(paginaPai, enderecoPaginaPai);
390 
391  promoverOParQueEstiverSobrando(
392  indiceDePromocao, paginaDeInsercao,
393  inseriuNaPaginaFilha, infoPai);
394 
395  if (voltouParaARaiz) trocarRaizPor(paginaPai);
396  }
397  }
398  }
399 
400  return infoPai;
401  }
