491  TIPO_DOS_DADOS excluir(TIPO_DAS_CHAVES &chave,
492  list<file_ptr_type>& pilhaDeEnderecos, list<int>& pilhaDeIndices)
493  {
494  TIPO_DOS_DADOS dadoExcluido;
495  // paginaFilha aponta para a última página carregada na descida da árvore
496  // (a página onde a chave foi encontrada ou alguma folha).
497  int indiceDaChave = paginaFilha->obterIndiceDeDescida(chave);
498 
499  // Checa se a chave realmente foi encontrada
500  if (paginaFilha->chaves[indiceDaChave] == chave)
501  {
502  dadoExcluido = paginaFilha->dados[indiceDaChave];
503  limparErro();
504 
505  if (paginaFilha->eUmaFolha())
506  {
507  paginaFilha->excluir(indiceDaChave, false, true);
508 
509  // Checa se o tamanho da paginaFilha antes da remoção era <= a 50%
510  if (paginaFilha->tamanho() + 1 <= numeroDeChavesPorPagina / 2)
511  {
512  // Obtém o índice do ponteiro na página pai que foi usado para
513  // chegar na paginaFilha
514  int indiceDeDescida = pilhaDeIndices.back();
515 
516  if (!pegarChaveEmprestada(indiceDeDescida))
517  {
518  fundirPaginas(indiceDeDescida);
519  pilhaDeEnderecos.pop_back();
520 
521  while (paginaPai->tamanho() < numeroDeChavesPorPagina / 2
522  && pilhaDeEnderecos.size() > 1)
523  {
524  swap(paginaPai, paginaFilha);
525  pilhaDeIndices.pop_back();
526  pilhaDeEnderecos.pop_back();
527  carregar(paginaPai, pilhaDeEnderecos.back());
528  fundirPaginas(pilhaDeIndices.back());
529  }
530 
531  if (pilhaDeEnderecos.size() == 1 && paginaPai->vazia())
532  trocarRaizPor(paginaFilha);
533  }
534  }
535 
536  paginaFilha->colocarNoArquivo(arquivo);
537  }
538 
539  else
540  {
541  trocarChavePorAntecessora(
542  indiceDaChave, pilhaDeEnderecos, pilhaDeIndices);
543  excluir(chave, pilhaDeEnderecos, pilhaDeIndices);
544  }
545  }
546 
547  else atribuirErro("A chave não foi encontrada");
548 
549  return dadoExcluido;
550  }

667  void inserir(TIPO_DAS_CHAVES &chave, TIPO_DOS_DADOS &dado,
668  list<file_ptr_type>& pilhaDeEnderecos,
669  list<int>& pilhaDeIndices)
670  {
671  // Após a função obterCaminhoDeDescida(), paginaFilha aponta para a última
672  // página carregada na descida da árvore (alguma folha).
673  int indiceDeInsercao = paginaFilha->obterIndiceDeDescida(chave);
674 
675  // Checa se a inserção teve sucesso
676  if (paginaFilha->inserir(chave, dado, indiceDeInsercao))
677  paginaFilha->colocarNoArquivo(arquivo);
678 
679  else // Inserção na página falhou, acontece quando ela está cheia.
680  {
681  pilhaDeEnderecos.pop_back(); // O último endereço pode ser descartado
682  bool voltouParaARaiz = pilhaDeEnderecos.empty();
683  int indiceDePromocao = pilhaDeIndices.back();
684 
685  pair<Pagina *, bool> infoPai(nullptr, false);
686  tratarPaginaCheia(chave, dado, indiceDePromocao, infoPai);
687 
688  if (voltouParaARaiz) trocarRaizPor(paginaPai);
689 
690  else // Inicia o processo de subida na árvore
691  {
692  // Quando uma página que é filha da atual promove um elemento para
693  // a atual ou para a irmã da atual, a variável paginaDeInsercao vai
694  // ter um ponteiro para a página que receber o elemento promovido.
695  Pagina *paginaDeInsercao = infoPai.first;
696  bool inseriuNaPaginaAtual;
697  file_ptr_type enderecoPaginaPai;
698 
699  // Checa se houve alguma promoção para a página atual
700  while (paginaDeInsercao != nullptr)
701  {
702  pilhaDeEnderecos.pop_back();
703  voltouParaARaiz = pilhaDeEnderecos.empty();
704  enderecoPaginaPai = voltouParaARaiz ?
705  constantes::ptrNuloPagina : pilhaDeEnderecos.back();
706 
707  pilhaDeIndices.pop_back();
708  indiceDePromocao = pilhaDeIndices.back();
709  inseriuNaPaginaAtual = infoPai.second;
710 
711  // As páginas que eram pais, na subida da árvore, são filhas
712  swap(paginaPai, paginaFilha);
713  swap(paginaIrmaPai, paginaIrma);
714 
715  if (voltouParaARaiz)
716  {
717  paginaPai->limpar();
718  paginaPai->ponteiros.push_back(constantes::ptrNuloPagina);
719  }
720 
721  else carregar(paginaPai, enderecoPaginaPai);
722 
723  promoverOParQueEstiverSobrando(
724  indiceDePromocao, paginaDeInsercao,
725  inseriuNaPaginaAtual, infoPai);
726 
727  if (voltouParaARaiz) trocarRaizPor(paginaPai);
728 
729  paginaDeInsercao = infoPai.first;
730  }
731  }
732  }
733  }
