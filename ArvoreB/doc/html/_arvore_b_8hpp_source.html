469  void inserir(TIPO_DAS_CHAVES &chave, TIPO_DOS_DADOS &dado,
470  list<file_ptr_type>& pilhaDeEnderecos,
471  list<int>& pilhaDeIndices)
472  {
473  // Após a função obterCaminhoDeDescida(), paginaFilha aponta para a última
474  // página carregada na descida da árvore (alguma folha).
475  int indiceDeInsercao = paginaFilha->obterIndiceDeDescida(chave);
476 
477  // Checa se a inserção teve sucesso
478  if (paginaFilha->inserir(chave, dado, indiceDeInsercao))
479  paginaFilha->colocarNoArquivo(arquivo);
480 
481  else // Inserção na página falhou, acontece quando ela está cheia.
482  {
483  pilhaDeEnderecos.pop_back(); // O último endereço pode ser descartado
484  bool voltouParaARaiz = pilhaDeEnderecos.empty();
485  int indiceDePromocao = pilhaDeIndices.back();
486 
487  pair<Pagina *, bool> infoPai(nullptr, false);
488  tratarPaginaCheia(chave, dado, indiceDePromocao, infoPai);
489 
490  if (voltouParaARaiz) trocarRaizPor(paginaPai);
491 
492  else // Inicia o processo de subida na árvore
493  {
494  // Quando uma página que é filha da atual promove um elemento para
495  // a atual ou para a irmã da atual, a variável paginaDeInsercao vai
496  // ter um ponteiro para a página que receber o elemento promovido.
497  Pagina *paginaDeInsercao = infoPai.first;
498  bool inseriuNaPaginaAtual;
499  file_ptr_type enderecoPaginaPai;
500 
501  // Checa se houve alguma promoção para a página atual
502  while (paginaDeInsercao != nullptr)
503  {
504  pilhaDeEnderecos.pop_back();
505  voltouParaARaiz = pilhaDeEnderecos.empty();
506  enderecoPaginaPai = voltouParaARaiz ?
507  constantes::ptrNuloPagina : pilhaDeEnderecos.back();
508 
509  pilhaDeIndices.pop_back();
510  indiceDePromocao = pilhaDeIndices.back();
511  inseriuNaPaginaAtual = infoPai.second;
512 
513  // As páginas que eram pais, na subida da árvore, são filhas
514  swap(paginaPai, paginaFilha);
515  swap(paginaIrmaPai, paginaIrma);
516 
517  if (voltouParaARaiz)
518  {
519  paginaPai->limpar();
520  paginaPai->ponteiros.push_back(constantes::ptrNuloPagina);
521  }
522 
523  else carregar(paginaPai, enderecoPaginaPai);
524 
525  promoverOParQueEstiverSobrando(
526  indiceDePromocao, paginaDeInsercao,
527  inseriuNaPaginaAtual, infoPai);
528 
529  if (voltouParaARaiz) trocarRaizPor(paginaPai);
530 
531  paginaDeInsercao = infoPai.first;
532  }
533  }
534  }
